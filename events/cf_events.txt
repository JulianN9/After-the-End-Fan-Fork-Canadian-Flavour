# Confederation Wars
# Called from 'on_crusade_creation'
# new_character is the targeted enemy
# ROOT is the religious head
# FROM is the attacked kingdom title
narrative_event = {
	id = canada.104
	title = EVTNAME200
	picture = GFX_evt_battle
	border = GFX_event_narrative_frame_war
	
	desc = {
		text = confeddisc.4
		trigger = {
			new_character = {
				NOT = { 
					religion_group = christian 
					religion_group = oldworld
					religion_group = muslim 
				}
			}
		}
	}
	desc = {
		text = confeddisc.4_oldworld
		trigger = {
			new_character = {
				OR = {
					religion_group = oldworld
					religion_group = muslim
				}
			}
		}
	}
	desc = {
		text = confeddisc.4_christian
		trigger = {
			new_character = {
				religion_group = christian
			}
		}
	}
	
	is_triggered_only = yes
	major = yes
	show_ROOT = yes
	hide_new = yes
	
	trigger = {
		can_call_crusade = yes
		ROOT = { religion = confederated }
	}
	
	option = {
		name = confedopta.4 # United under the Leaf!
		trigger = { religion = ROOT }
	}
	option = {
		name = EVTOPTB200 # No concern of ours
		trigger = {
			NOT = { religion = ROOT }
			new_character = {
				NOT = { religion = PREV }
				NOT = { character = PREV }
			}
		}
	}
	option = {
		name = EVTOPTD200 # To Arms!
		trigger = {
			new_character = {
				character = PREV
			}
		}
	}
	option = {
		name = EVTOPTE200 # Terrible News!
		trigger = {
			new_character = {
				religion = PREV
				NOT = { character = PREV }
			}
		}
	}
}

# Convert or Not?
narrative_event = {
	id = canada.105
	title = EVTNAMEcanada.1
	desc = EVTDESCcanada.1
	
	picture = GFX_evt_canada_confederation
	
	is_triggered_only = yes
	
	option = { # Yes
		name = EVTOPTAcanada.2
		if = {
			limit = {
				OR = {
					religion = high_church
					religion = nonconformist
				}
			}
			religion = united
			add_trait = english_rites
		}
		if = {
			limit = {
				OR = {
					religion = ursuline
					religion = angeline
				}
			}
			religion = united
			add_trait = french_rites
		}
		any_courtier = {
			if = {
				limit = {
					OR = {
						religion = high_church
						religion = nonconformist
					}
				}
				religion = united
				add_trait = english_rites
			}
			if = {
				limit = {
					OR = {
						religion = ursuline
						religion = angeline
					}
				}
				religion = united
				add_trait = french_rites
			}
		}
		capital_scope = {
			religion = united
		}
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.25
				government = theocracy_government
			}
			modifier = {
				factor = 0
				government = order_government
			}
			modifier = {
				factor = 0
				trait = zealous
			}
		}
	}
	option = { # No
		name = EVTOPTBcanada.2
		ai_chance = { factor = 25 }
	}
}
